cmake_minimum_required(VERSION 3.15)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Release" CACHE STRING "Choose the type of build, options are: Debug, Release, or MinSizeRel." FORCE)
    message(STATUS "CMAKE_BUILD_TYPE not set, defaulting to Release.")
endif()

set(CMAKE_MODULE_PATH "${CMAKE_MODULE_PATH}" "${CMAKE_SOURCE_DIR}/cmake")

if(NOT BUILD_NUMBER)
    set(BUILD_NUMBER 0)
endif()
set(CPU_FEATURES_VERSION 1.0.${BUILD_NUMBER})

project(cpufeatures VERSION "${CPU_FEATURES_VERSION}" LANGUAGES CXX C)

message(STATUS "Generator .............. ${CMAKE_GENERATOR}")
message(STATUS "Build Type ............. ${CMAKE_BUILD_TYPE}")

find_package(Threads)

add_compile_definitions(SK_USER_CONFIG_HEADER="${CMAKE_SOURCE_DIR}/SkUserConfig.h" _GNU_SOURCE)

add_executable(cpufeatures cpu-features.c main.cpp)
target_compile_options(cpufeatures PUBLIC -Wall -Wextra -Werror)
target_link_libraries(cpufeatures ${CMAKE_THREAD_LIBS_INIT} -ldl)
